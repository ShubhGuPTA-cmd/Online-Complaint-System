CREATE DATABASE OnlineComplaintSystem;
USE OnlineComplaintSystem;
2. Roles Table
CREATE TABLE Roles (
    roleId INT PRIMARY KEY AUTO_INCREMENT,
    roleType VARCHAR(50) NOT NULL
);
3. User Table
CREATE TABLE User (
    userId INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    alt_email VARCHAR(255),
    password VARCHAR(255) NOT NULL,
    roleId INT,
    FOREIGN KEY (roleId) REFERENCES Roles(roleId)
);
4. User Phone Table
CREATE TABLE User_Phone_No (
    phone_no BIGINT NOT NULL,
    userId INT,
    FOREIGN KEY (userId) REFERENCES User(userId)
);
5. Complainant Table
CREATE TABLE Complainant (
    nationalID BIGINT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    address VARCHAR(255)
);
6. Complainant Gender Table
CREATE TABLE Complainant_Gender (
    gender VARCHAR(10),
    nationalID BIGINT,
    FOREIGN KEY (nationalID) REFERENCES Complainant(nationalID)
);
7. Complainant Mobile Table
CREATE TABLE Complainant_Mobile (
    mobile BIGINT NOT NULL,
    nationalID BIGINT,
    FOREIGN KEY (nationalID) REFERENCES Complainant(nationalID)
);
8. Governorates Table
CREATE TABLE Governorates (
    governID INT PRIMARY KEY AUTO_INCREMENT,
    governName VARCHAR(255) NOT NULL
);
9. Regions Table
CREATE TABLE Regions (
    id INT PRIMARY KEY AUTO_INCREMENT,
    regionName VARCHAR(255) NOT NULL,
    governID INT,
    FOREIGN KEY (governID) REFERENCES Governorates(governID)
);
10. Marital Status Table
CREATE TABLE MaritalStatus (
    id INT PRIMARY KEY AUTO_INCREMENT,
    type VARCHAR(50) NOT NULL
);
11. Complaint Table
CREATE TABLE Complaint (
    complaintID INT PRIMARY KEY AUTO_INCREMENT,
    complaintNumber VARCHAR(50) UNIQUE,
    complaintDesc TEXT NOT NULL,
    complaintDate DATE NOT NULL,
    nationalID BIGINT,
    FOREIGN KEY (nationalID) REFERENCES Complainant(nationalID)
);
12. Status Table
CREATE TABLE Status (
    id INT PRIMARY KEY AUTO_INCREMENT,
    statusType VARCHAR(50) NOT NULL
);
13. Complaint Status Table
CREATE TABLE ComplaintStatus (
    statusId INT PRIMARY KEY AUTO_INCREMENT,
    complaintID INT,
    userId INT,
    id INT,
    FOREIGN KEY (complaintID) REFERENCES Complaint(complaintID),
    FOREIGN KEY (userId) REFERENCES User(userId),
    FOREIGN KEY (id) REFERENCES Status(id)
);
14. FollowUp Table
CREATE TABLE FollowUp (
    followupID INT PRIMARY KEY AUTO_INCREMENT,
    followupDate DATE,
    complaintID INT,
    FOREIGN KEY (complaintID) REFERENCES Complaint(complaintID)
);
15. Complaint Causes Table
CREATE TABLE ComplaintCauses (
    causeID INT PRIMARY KEY AUTO_INCREMENT,
    causeName VARCHAR(255) NOT NULL,
    complaintID INT,
    FOREIGN KEY (complaintID) REFERENCES Complaint(complaintID)
);
16. Complaint Actions Table
CREATE TABLE ComplaintActions (
    complaintActionID INT PRIMARY KEY AUTO_INCREMENT,
    actions TEXT NOT NULL,
    complaintID INT,
    FOREIGN KEY (complaintID) REFERENCES Complaint(complaintID)
);
17. Questions Table
CREATE TABLE Questions (
    qID INT PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL
);
18. Answers Table
CREATE TABLE Answers (
    ansID INT PRIMARY KEY AUTO_INCREMENT,
    qID INT,
    FOREIGN KEY (qID) REFERENCES Questions(qID)
);
19. Site Survey Table
CREATE TABLE SiteSurvey (
    surveyID INT PRIMARY KEY AUTO_INCREMENT,
    date DATE,
    nationalID BIGINT,
    FOREIGN KEY (nationalID) REFERENCES Complainant(nationalID)
);
